279        responses = data["responses"]
280        remarks = ""

282        cur = con.cursor()

284        query = "INSERT INTO APPLICATIONS (STUDENT, STATUS, POSTING_ID, REMARKS, CREATED_AT, UPDATED_AT) VALUES (:1,:2,:3,:4,SYSTIMESTAMP,SYSTIMESTAMP) RETURNING APPLICATION_ID into :5"

287
288        # Get ID of inserted

290
291        # Insert responses into database
292        for i in range(len(responses)):

295            params = [responses[i]['answer'], responses[i]['question_id'], inserted_id]
296            cur.execute(query, params)
297

301        )
